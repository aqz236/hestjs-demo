app.controller.ts
// app.controller.ts
import { Controller, Get, Inject } from '@hestjs/core';
import { AppService } from './app.service';
@Controller()
export class AppController {
constructor(
@Inject() private appService: AppService
) {}
@Get('/')
async getHello() {
return this.appService.getHello();
}
}

Why Choose HestJS?

HestJS combines the best of NestJS's developer experience with Hono's performance, giving you a modern framework that scales with your needs.

Blazing Fast

Built on Hono, delivering exceptional performance with minimal overhead for high-throughput applications.

Type Safe

Full TypeScript support with strict typing, catching errors at compile time for robust applications.

Developer Experience

Familiar NestJS patterns with modern tooling, hot reload, and excellent debugging capabilities.

Modular Architecture

Clean, modular design with dependency injection powered by TSyringe for maintainable code.

CQRS Support

Built-in Command Query Responsibility Segregation pattern for scalable application architecture.

Production Ready

Battle-tested patterns with comprehensive logging, validation, and API documentation tools.
